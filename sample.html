function mixin(target, source) {
        Object.assign(target, source);
      }
      
      class ClassA {
        methodA() {
          console.log("Method A");
        }
      }
      
      class ClassB {
        methodB() {
          console.log("Method B");
        }
      }
      
      class ClassC {
        constructor() {
         this.val = "hi"
        }
        methodC()
        {
          console.log("hi1");
        }
      }
      
      mixin(ClassC, ClassA.prototype); // Correctly mixin to the prototype
      mixin(ClassC, ClassB.prototype);
      
      const instanceC = new ClassC();
      instanceC.methodA(); // Output: Method A
      instanceC.methodB(); // Output: Method B
